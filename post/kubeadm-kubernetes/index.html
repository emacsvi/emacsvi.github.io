利用kubeadm部署kubernetes

2019年4月16日
| kubernetes

利用kubeadm部署教程。

部署整个过程

1. 准备vagrant集群文件

# -*- mode: ruby -*-
# vi: set ft=ruby :

Vagrant.configure("2") do |config|

# 设置master虚拟机环境
  config.vm.define "master" do |master|
    # 设置虚拟机的box
    master.vm.box = "centos/7"
    # 设置虚拟机的主机名
    master.vm.hostname = "master"
    # 设置虚拟机的网络
    master.vm.network "private_network", ip: "172.27.128.150"
    # VirtaulBox相关设置
    master.vm.provider "virtualbox" do |vb|
      vb.memory = 2048
      vb.cpus = 2
      vb.name = "master"
    end
    # master上shell脚本
    # master.vm.provision "shell", path: "master.sh"
  end

# 设置node1虚拟机环境
  config.vm.define "node1" do |node1|
    # 设置虚拟机的box
    node1.vm.box = "centos/7"
    # 设置虚拟机的主机名
    node1.vm.hostname = "node1"
    # 设置虚拟机的网络
    node1.vm.network "private_network", ip: "172.27.128.149"
    # VirtaulBox相关设置
    node1.vm.provider "virtualbox" do |vb|
      vb.memory = 1024
      vb.cpus = 2
    end
    # node1上shell脚本
    # node1.vm.provision "shell", path: "node1.sh"
  end

# 设置node2虚拟机环境
  config.vm.define "node2" do |node2|
    # 设置虚拟机的box
    node2.vm.box = "centos/7"
    # 设置虚拟机的主机名
    node2.vm.hostname = "node2"
    # 设置虚拟机的网络
    node2.vm.network "private_network", ip: "172.27.128.148"
    # VirtaulBox相关设置
    node2.vm.provider "virtualbox" do |vb|
      vb.memory = 1024
      vb.cpus = 1
    end
    # node1上shell脚本
    # node1.vm.provision "shell", path: "node1.sh"
  end


  config.vm.synced_folder ".", "/vagrant", type: "nfs", nfs_udp: false

  # master和node公共脚本
  config.vm.provision "shell", path: "./common.sh"

  # ssh配置
  config.ssh.username = "vagrant"
  config.ssh.private_key_path = "~/.vagrant.d/insecure_private_key"
  config.ssh.insert_key = false
end

2. 准备初始化系统的安装脚本

这个脚本就是在vagrant up的时候创建好虚拟机之后所执行的comm.sh脚本。里面将源信息，常用的一些软件都安装在里面，是不是很方便。

echo "Please set utf8 if Chinese garbled under minGW"
echo "=====执行公共脚本====="

echo "允许root用户远程公钥认证"
mkdir -p /root/.ssh
chmod 700 /root/.ssh
cp /home/vagrant/.ssh/authorized_keys /root/.ssh/

echo "将yum的base源设置为aliyun"
if [ -z "`sed -n -e '/aliyun.*CentOS-7/p' /etc/yum.repos.d/CentOS-Base.repo`" ]; then
mv /etc/yum.repos.d/CentOS-Base.repo /etc/yum.repos.d/CentOS-Base.repo.bak
curl -o /etc/yum.repos.d/CentOS-Base.repo \
    http://mirrors.aliyun.com/repo/Centos-7.repo
fi

yum install -y epel-release
yum install -y conntrack ipvsadm ipset jq iptables curl sysstat libseccomp
/usr/sbin/modprobe ip_vs

echo "禁用fastmirror插件"
sed -i 's/enabled=1/enabled=0/' /etc/yum/pluginconf.d/fastestmirror.conf

echo "yum安装net-tools（centos7.5的box没有net-bools）"
yum install -y net-tools

echo "yum安装vim"
yum install -y vim
echo "alias vi=vim" >> /root/.bashrc

echo "永久关闭firewalld"
systemctl disable firewalld

echo "设置CST时区"
timedatectl set-timezone Asia/Shanghai

echo "重启依赖系统时间的服务"
systemctl restart rsyslog
systemctl restart crond

echo "load ipvs model"
cat > /etc/sysconfig/modules/ipvs.modules <<EOF
#!/bin/bash
modprobe -- ip_vs
modprobe -- ip_vs_rr
modprobe -- ip_vs_wrr
modprobe -- ip_vs_sh
modprobe -- nf_conntrack_ipv4
EOF

yum install -y wget

echo "install kubelet"
cat << EOF > /etc/yum.repos.d/kubernetes.repo
[kubernetes]
name=Kubernetes
baseurl=https://mirrors.aliyun.com/kubernetes/yum/repos/kubernetes-el7-x86_64/
enabled=1
gpgcheck=1
repo_gpgcheck=1
gpgkey=https://mirrors.aliyun.com/kubernetes/yum/doc/yum-key.gpg https://mirrors.aliyun.com/kubernetes/yum/doc/rpm-package-key.gpg
EOF

wget http://mirrors.aliyun.com/repo/Centos-7.repo -O /etc/yum.repos.d/CentOS-Base.repo
wget http://mirrors.aliyun.com/repo/epel-7.repo -O /etc/yum.repos.d/epel.repo
wget https://mirrors.aliyun.com/docker-ce/linux/centos/docker-ce.repo -O /etc/yum.repos.d/docker-ce.repo

chmod 755 /etc/sysconfig/modules/ipvs.modules && bash /etc/sysconfig/modules/ipvs.modules && lsmod | grep -e ip_vs -e nf_conntrack_ipv4

# 安装常用软件
yum install -y wget curl conntrack-tools vim net-tools telnet tcpdump bind-utils socat ntp kmod ceph-common dos2unix
yum install -y kubelet kubeadm kubectl ipvsadm ipset docker-ce


echo 'disable selinux'
setenforce 0
sed -i 's/=enforcing/=disabled/g' /etc/selinux/config

echo 'enable iptable kernel parameter'
cat >> /etc/sysctl.conf <<EOF
net.ipv4.ip_forward=1
EOF
sysctl -p

systemctl stop firewalld
systemctl disable firewalld
iptables -F && iptables -X && iptables -F -t nat && iptables -X -t nat
iptables -P FORWARD ACCEPT

modprobe br_netfilter
cat > kubernetes.conf <<EOF
net.bridge.bridge-nf-call-iptables=1
net.bridge.bridge-nf-call-ip6tables=1
net.ipv4.ip_forward=1
net.ipv4.tcp_tw_recycle=0
vm.swappiness=0 # 禁止使用 swap 空间，只有当系统 OOM 时才允许使用它
vm.overcommit_memory=1 # 不检查物理内存是否够用
vm.panic_on_oom=0 # 开启 OOM
fs.inotify.max_user_instances=8192
fs.inotify.max_user_watches=1048576
fs.file-max=52706963
fs.nr_open=52706963
net.ipv6.conf.all.disable_ipv6=1
net.netfilter.nf_conntrack_max=2310720
EOF
cp kubernetes.conf  /etc/sysctl.d/kubernetes.conf
sysctl -p /etc/sysctl.d/kubernetes.conf


echo 'set host name resolution'
cat >> /etc/hosts <<EOF
172.27.128.150 master master.emacsvi.com
172.27.128.149 node1 node1.emacsvi.com
172.27.128.148 node2 node2.emacsvi.com
EOF

cat /etc/hosts

echo 'set nameserver'
echo "nameserver 8.8.8.8">/etc/resolv.conf
cat /etc/resolv.conf

echo 'disable swap'
swapoff -a
sed -i '/swap/s/^/#/' /etc/fstab

cat <<EOF >  /etc/sysctl.d/k8s.conf
net.bridge.bridge-nf-call-ip6tables = 1
net.bridge.bridge-nf-call-iptables = 1
EOF
sysctl --system

mkdir -p /etc/docker
cat > /etc/docker/daemon.json <<EOF
{
 "registry-mirrors": ["https://registry.docker-cn.com"]
}
EOF


#启动docker
systemctl enable docker && systemctl start docker

echo 'KUBELET_EXTRA_ARGS="--fail-swap-on=false"' > /etc/sysconfig/kubelet
#设置kubelet开机自启动
systemctl enable kubelet

yum install -y bash-completion*
##写入环境变量
source <(kubectl completion bash)
echo "source <(kubectl completion bash)" >> /root/.bashrc

yum install -y rpcbind
systemctl enable rpcbind.service
systemctl start rpcbind

# 这个命令在里面进行安装
docker load -i k8s-1.14.0.tar.gz

3.然后启动各个机器

vagrant up即可启动这个集群，里面有3个节点。分别是master,node1,node2这三个节点。

4.配置

在所有节点上必须要手动安装image文件，因为被墙了出不去。所以最后想办法打包下载好之后再导入到机器之中。

# master上
vagrant ssh master
sudo -i
passwd root
vim /etc/ssh/sshd_config
ssh-keygen
ssh-copy-id root@node1
ssh-copy-id root@node2
systemctl restart sshd

docker load -i k8s.1.14.0.tar.gz # 安装image

# 这几个镜像的事情我当时也是自作聪明的以为只需要flannel:v0.11.0-amd64即可。谁知道连arm s390x ppc64le这些凡是在kube-flannel.yaml文件内容里面有的，都需要，不然就启动不成功

# 查看kubelet配置文件里面的内容是不是增加了一些错误忽略
cat /etc/sysconfig/kubelet

# 初始化配置，这里一定要指定--apiserver-advertise-address=172.27.128.150这个地址真正物理出外网的地址，否则可以自动会选出错导致slave节点连接不上。
kubeadm init --apiserver-advertise-address=172.27.128.150 --kubernetes-version=v1.14.0 --pod-network-cidr=10.244.0.0/16  --service-cidr=10.96.0.0/12 --ignore-preflight-errors=all --token-ttl 0 | tee ~/kubeadm.init.log

# 假设上面的配置你写错了。需要重新reset一次。那么运行如下命令
kubeadm reset
iptables -F && iptables -t nat -F && iptables -t mangle -F && iptables -X
ipvsadm --clear
systemctl restart kubelet
systemctl status kubelet
# 上面这一堆都是重置命令

mkdir -p $HOME/.kube
cp -i /etc/kubernetes/admin.conf $HOME/.kube/config
chown $(id -u):$(id -g) $HOME/.kube/config
kubectl get nodes
# 安装flannel
kubectl apply -f https://raw.githubusercontent.com/coreos/flannel/master/Documentation/kube-flannel.yml
kubectl get pods -n kube-system
kubectl get nodes


# 在node节点上
docker load -i k8s-1.14.0.tar.gz
docker images
cat /etc/sysconfig/kubelet

在vagrant集群启动多机器时会遇到nfs出错的问题。

报错信息内容大致这样：

The following SSH command responded with a non-zero exit status.
Vagrant assumes that this means the command failed!

mount -o vers=3 172.27.128.1:/Volumes/golang/coding/kubernetes/kubeadm-flannel /vagrant

Stdout from the command:



Stderr from the command:

A dependency job for rpc-statd.service failed. See 'journalctl -xe' for details.
mount.nfs: rpc.statd is not running but is required for remote locking.
mount.nfs: Either use '-o nolock' to keep locks local, or start statd.
mount.nfs: an incorrect mount option was specified

这是因为需要开启rpcbind服务才可以：

yum install -y rpcbind
systemctl enable rpcbind.service
systemctl start rpcbind

另外在vagrantfile之中执行脚本的路径是宿主机的路径。而不是虚拟机里面的路径哦。

# master和node公共脚本
  config.vm.provision "shell", path: "./common.sh"

安装docker,kubernetes相关的bin文件

cd /etc/yum.repos.d/
wget https://mirrors.aliyun.com/docker-ce/linux/centos/docker-ce.repo
cat >> /etc/yum.repos.d/kubernetes.repo <<EOF
[kubernetes]
name=Kubernetes
baseurl=https://mirrors.aliyun.com/kubernetes/yum/repos/kubernetes-el7-x86_64/
enabled=1
gpgcheck=0
EOF
yum install -y docker-ce kubelet kubectl kubeadm
